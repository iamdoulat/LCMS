// Temporary helper functions with debug logging
const getLeaveStatus = (date: Date, leaveApplications: any[]) => {
    const found = leaveApplications.find(leave => {
        const start = parseISO(leave.fromDate);
        const end = parseISO(leave.toDate);
        const result = isWithinInterval(date, { start, end });
        
        // Debug logging for first day of month
        if (format(date, 'd') === '1' && result) {
            console.log('âœ… Leave found:', {
                date: format(date, 'yyyy-MM-dd'),
                leaveId: leave.id,
                fromDate: leave.fromDate,
                toDate: leave.toDate,
                status: leave.status
            });
        }
        
        return result;
    });
    return found;
};

const getHoliday = (date: Date, holidays: any[]) => {
    const found = holidays.find(h => {
        const start = parseISO(h.fromDate);
        const end = parseISO(h.toDate || h.fromDate);
        const result = isWithinInterval(date, { start, end });
        
        // Debug logging
        if (format(date, 'd') === '1' && result) {
            console.log('ðŸŽ‰ Holiday found:', {
                date: format(date, 'yyyy-MM-dd'),
                name: h.name,
                fromDate: h.fromDate,
                toDate: h.toDate
            });
        }
        
        return result;
    });
    return found;
};
